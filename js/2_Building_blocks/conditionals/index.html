条件文 swtich

          
            switch ( 式 ) {
              case 選択肢1:
                このコードを実行する
                break;
            
              case 選択肢2:
                代わりにこのコードを実行する
                break;
                
              // 以下に選択肢を好きなだけ並べる
            
              default:
                既定でこのコードを実行する
            }

条件文 ３項演算子

          
            ( 条件式 ) ? こちらのコードを実行する : 代わりにこちらのコードを実行する
            let greeting = ( isBirthday ) ? 'スミスさん、誕生日おめでとうございます！良い一日を。' : 'スミスさんおはようございます。';
