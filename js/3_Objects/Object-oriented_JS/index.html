コンストラクターとオブジェクトインスタンス


  function Person(name) {
    this.name = name;
    this.greeting = function() {
      alert('Hi! I\'m' + this.name + '.');
    };
  }



let person1 = new Person('Bob');
let person1 = new Person('Sarah');
